# Supabase Schema Plan

Planned tables:
- `users`
- `telegram_users`
- `payment_sessions`
- `invoices`
- `payments`
- `ledger_entries`
- `entitlements`
- `review_requests`
- `audit_logs`
- `risk_checks`
- `contract_scans`
- `contract_reports`
- `contract_risk_signals`
- `contract_sources`
- `approval_checks`
- `scan_jobs`
- `reported_wallets`
- `scam_reports`
- `agents`
- `guarded_transactions`
- `disputes`
- `data_sources`
- `risk_model_versions`

MVP rule: the app must not require Supabase to be configured in order to run. Supabase persistence should extend the existing no-crash audit and payment-session patterns.

Sensitive-data rule: do not store seed phrases, private keys, wallet passwords, or raw documents that are not required for the paid service.

## Production-ready starter tables

Use snake_case column names. Keep row-level security policies business-appropriate before exposing any authenticated client features. Server writes should use `SUPABASE_SERVICE_ROLE_KEY` only in Vercel server-side env vars.

```sql
create table if not exists payment_sessions (
  id text primary key,
  telegram_chat_id bigint,
  telegram_user_id bigint,
  telegram_username text,
  selected_product_id text not null,
  payment_provider text not null,
  status text not null,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists invoices (
  id text primary key,
  session_id text references payment_sessions(id),
  product_id text not null,
  amount_usd numeric not null,
  amount_ghs numeric,
  provider text not null,
  status text not null,
  external_reference text,
  created_at timestamptz not null default now()
);

create table if not exists payments (
  id text primary key,
  invoice_id text,
  session_id text,
  product_id text,
  provider text not null,
  status text not null,
  reference text,
  crypto_tx_hash text,
  amount_received text,
  currency text,
  channel text,
  created_at timestamptz not null default now()
);

create table if not exists ledger_entries (
  id text primary key,
  payment_id text,
  session_id text,
  direction text not null,
  amount text not null,
  currency text not null,
  provider text not null,
  description text not null,
  created_at timestamptz not null default now()
);

create table if not exists review_requests (
  id text primary key,
  session_id text not null,
  product_id text not null,
  status text not null default 'paid',
  subject_type text not null,
  subject_value text not null,
  network text,
  payment_reference text,
  crypto_tx_hash text,
  telegram_handle text,
  contact_email text,
  context text,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists audit_logs (
  id bigint generated by default as identity primary key,
  event_type text not null,
  actor_type text not null,
  actor_id text,
  command text,
  session_id text,
  payment_id text,
  metadata jsonb not null default '{}'::jsonb,
  created_at timestamptz not null default now()
);

-- Planned contract analyzer tables. Keep source storage minimal and avoid secrets.
create table if not exists contract_scans (
  id text primary key,
  telegram_chat_id bigint,
  telegram_user_id bigint,
  chain text not null,
  contract_address text not null,
  scan_mode text not null,
  risk_score integer,
  risk_level text,
  source_status text,
  created_at timestamptz not null default now()
);

create table if not exists contract_reports (
  id text primary key,
  scan_id text references contract_scans(id),
  product_id text,
  status text not null,
  summary text,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists contract_risk_signals (
  id bigint generated by default as identity primary key,
  scan_id text references contract_scans(id),
  signal_id text not null,
  severity text not null,
  label text not null,
  score integer not null,
  created_at timestamptz not null default now()
);

create table if not exists contract_sources (
  id text primary key,
  chain text not null,
  contract_address text not null,
  source_status text not null,
  explorer_url text,
  source_hash text,
  created_at timestamptz not null default now()
);

create table if not exists approval_checks (
  id text primary key,
  chain text not null,
  spender_contract text not null,
  owner_wallet text,
  allowance_context text,
  risk_score integer,
  created_at timestamptz not null default now()
);

create table if not exists scan_jobs (
  id text primary key,
  scan_id text,
  job_type text not null,
  status text not null,
  attempts integer not null default 0,
  last_error text,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);
```
